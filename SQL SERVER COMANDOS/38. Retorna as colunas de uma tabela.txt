SELECT 
    COLUNAS.NAME AS COLUNA,
    TIPOS.NAME AS TIPO,
    COLUNAS.LENGTH AS TAMANHO,
    COLUNAS.ISNULLABLE AS EH_NULO
 
FROM 
    SYSOBJECTS AS TABELAS,
    SYSCOLUMNS AS COLUNAS,
    SYSTYPES   AS TIPOS
WHERE 
    -- JOINS 
    TABELAS.ID = COLUNAS.ID
    AND COLUNAS.USERTYPE = TIPOS.USERTYPE
    AND TABELAS.NAME = 'CLIFOR'
