SELECT IDLAN
FROM FLAN
WHERE CODCOLIGADA = 1
AND NFOUDUP <> 1
AND CODTB2FLX=023
AND (NUMBLOQUEIOS = 0 OR NUMBLOQUEIOS IS NULL)
AND(BAIXAAUTORIZADA IS NULL OR BAIXAAUTORIZADA <> 0)
AND PAGREC = 1 AND CODCOLCFO = 1
AND STATUSLAN = 1
AND NOT EXISTS (SELECT * FROM FRELLAN R (NOLOCK) WHERE (FLAN.IDLAN =
R.IDLAN OR FLAN.IDLAN = R.IDLANREL) AND R.TIPOREL = 2 AND R.CODCOLIGADA =
1) AND CODTDO IN (SELECT CODTDO FROM FTDO WHERE CODCOLIGADA = 1 AND
EDEVOLUCAO = 2 AND CODTDO='ADI') AND VALORADIANTAMENTO = 0 AND
VALORDEVOLUCAO = 0
