--LISTA LOTES NEGATIVOS
SELECT DISTINCT TPRD.CODIGOREDUZIDO,TLOTEPRDLOC.* 
FROM TMOV,TITMMOV,TPRD,TLOTEPRD,TITMLOTEPRD,TLOTEPRDLOC
WHERE TMOV.IDMOV=TITMMOV.IDMOV AND TITMMOV.IDPRD=TPRD.IDPRD AND TITMMOV.IDMOV=TITMLOTEPRD.IDMOV AND TITMMOV.NSEQITMMOV=TITMLOTEPRD.NSEQITMMOV
AND TITMLOTEPRD.IDLOTE=TLOTEPRD.IDLOTE AND TLOTEPRDLOC.IDLOTE=TLOTEPRD.IDLOTE AND (TLOTEPRDLOC.SALDOFISICO2 < 0 OR TLOTEPRDLOC.SALDOFISICO5 < 0)
