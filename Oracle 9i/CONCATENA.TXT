CREATE OR REPLACE FUNCTION CONCATENA ( QTD IN NUMBER )
RETURN VARCHAR2 IS RET VARCHAR2(255);

BEGIN
RET:='0000001';     
FOR I IN 1..QTD LOOP
  RET := RET || LPAD(I+1, 7 ,'0');
END LOOP;
  RET := RET || LPAD(QTD+2, 7 ,'0');
RETURN RET;
    
END CONCATENA; 

SELECT CONCATENA(2) FROM DUAL;
